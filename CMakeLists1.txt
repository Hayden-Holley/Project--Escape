cmake_minimum_required(VERSION 3.14)
project(SFMLGame)

set(CMAKE_CXX_STANDARD 17)

if(WIN32)
    set(SFML_STATIC_LIBRARIES TRUE)
    set(CMAKE_EXE_LINKER_FLAGS "-static")
endif()

set(SFML_DIR "C:/SFML-2.5.1/lib/cmake/SFML")
find_package(SFML 2.5.1 COMPONENTS system window graphics network audio REQUIRED)

set(SOURCE_FILES
        src/main.cpp src/Game.cpp src/Game.h src/FreeFuncs.cpp
        src/ImageLayer.cpp src/LayerGroup.cpp src/Map.cpp src/miniz.c src/miniz.h src/Object.cpp src/ObjectGroup.cpp
        src/ObjectTypes.cpp src/Property.cpp src/TileLayer.cpp src/Tileset.cpp src/detail/pugixml.cpp src/GameObject.h "src/Tile.h" src/GameObject.cpp "src/Tile.cpp" src/Tilemap.cpp src/Tilemap.h src/Character.cpp src/Character.h src/Client.cpp src/Client.h src/Server.cpp src/Server.h src/PacketOverrides.cpp src/PacketOverrides.h src/Enemy.cpp src/Enemy.h)

add_executable(SFMLGame ${SOURCE_FILES})

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

set(CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake_modules")

target_link_libraries (SFMLGame sfml-graphics sfml-window sfml-system sfml-network sfml-audio)
file(COPY ${CMAKE_CURRENT_SOURCE_DIR}/Data/ DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/Data/)
